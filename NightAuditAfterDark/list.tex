\begin{itemize}
    \item Check in remaining arrivals\newline 
    \textit{..if unable to authorize, cancel remaining arrivals}

    \begin{itemize}
        \item Shift#
        \item Date
        \item Name
        \item Amount of cash drop

    \item Sign the deposit log
    \item Drop the envelope into the safe
\end{itemize}
